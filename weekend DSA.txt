def is_leap(year):
    leap = False
 
    not_leap = [1800,1900,2100,2200,2300]
    
    if year%4 == 0 and year not in not_leap:
        leap = True
    
    if year%100 == 0:
        if year%400 ==0:
            leap = True
        else:
            leap = False 
    # Write your logic here
